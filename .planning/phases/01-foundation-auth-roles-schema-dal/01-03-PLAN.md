    import 'server-only'
    import { Resend } from 'resend'

    const resend = new Resend(process.env.RESEND_API_KEY!)

    export type SendEmailArgs = { to: string | string[]; subject: string; html: string }

    export async function sendEmail({ to, subject, html }: SendEmailArgs) {
      return resend.emails.send({
        from: process.env.EMAIL_FROM!,
        to,
        subject,
        html,
      })
    }

| T-01-13 | Spoofing | unverified sender domain / spoofed from | mitigate | `from` is fixed to process.env.EMAIL_FROM (a verified Resend sender), never caller-supplied |
