```typescript
// auth.config.ts  (EDGE-SAFE: NO bcrypt, NO db imports)
import type { NextAuthConfig } from "next-auth"

export const authConfig = {
  pages: { signIn: "/sign-in" },
  session: { strategy: "jwt" },
  callbacks: {
    authorized({ auth, request }) {
      const isLoggedIn = !!auth?.user
      const path = request.nextUrl.pathname
      const isPublic = ["/sign-in", "/sign-up", "/reset-password", "/verify-email", "/"].some(p => path === p || path.startsWith("/api/auth"))
      if (isPublic) return true
      return isLoggedIn // unauthenticated -> redirected to pages.signIn
    },
    jwt({ token, user }) {
      if (user) { token.id = user.id; token.role = (user as { role?: string }).role }
      return token
    },
    session({ session, token }) {
      if (session.user) {
        session.user.id = token.id as string
        session.user.role = token.role as "factory_pm" | "site_pm" | "super_admin"
      }
      return session
    },
  },
  providers: [], // providers added in auth.ts (Credentials needs bcrypt+db, not edge-safe)
} satisfies NextAuthConfig
```

```typescript
// auth.ts  (Node runtime: bcrypt + db allowed)
import NextAuth from "next-auth"
import Credentials from "next-auth/providers/credentials"
import bcrypt from "bcryptjs"
import { eq } from "drizzle-orm"
import { authConfig } from "@/auth.config"
import { db } from "@/db"
import { users } from "@/db/schema"

export const { handlers, auth, signIn, signOut } = NextAuth({
  ...authConfig,
  providers: [
    Credentials({
      credentials: { email: {}, password: {} },
      async authorize(creds) {
        const email = String(creds?.email ?? "").toLowerCase().trim()
        const password = String(creds?.password ?? "")
        if (!email || !password) return null
        const user = await db.query.users.findFirst({ where: eq(users.email, email) })
        if (!user?.hashedPassword) return null
        const ok = await bcrypt.compare(password, user.hashedPassword)
        if (!ok) return null
        return { id: user.id, email: user.email, name: user.name, role: user.role }
      },
    }),
  ],
})
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser -> Credentials authorize() | untrusted email/password crosses here |
| browser -> protected route | NextAuth JWT cookie presented; may be absent/forged |
| server component / action -> DAL | every authoritative check goes through await auth() |
| edge proxy -> request | optimistic only; never the authoritative gate |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-05 | Spoofing | credential stuffing / password guessing | mitigate | authorize() uses bcrypt.compare against a bcrypt hash; returns null (generic) on any failure — no user-existence signal |
| T-01-06 | Tampering | JWT/session cookie tampering | mitigate | NextAuth signs the JWT with AUTH_SECRET; `await auth()` verifies the signature before any role is trusted |
| T-01-07 | Information Disclosure | AUTH_SECRET exposure | mitigate | AUTH_SECRET read only from process.env in server/edge config; never NEXT_PUBLIC_; not logged |
| T-01-08 | Information Disclosure | hashedPassword leaking to client | mitigate | authorize() returns only { id, email, name, role } (no hash); DAL/session never carry the hash |
| T-01-09 | Elevation of Privilege | bypassing proxy by calling a route/action directly | mitigate | proxy is optimistic only; DAL (await auth() + requireRole/requireOwnerOrAdmin) is the authoritative gate in every layout/action |
| T-01-10 | Tampering | bcrypt cost too low | mitigate | use bcryptjs default cost (10) on hashing (Plan 04); compare is constant-work; acceptable for internal app v1 |
| T-01-11 | Denial of Service | bcrypt/db work in edge proxy | mitigate | proxy.ts imports only edge-safe auth.config.ts (no bcrypt/db) — grep-asserted |
</threat_model>
