    ```typescript
    import 'server-only'
    import { randomBytes, createHash } from 'node:crypto'
    import { eq } from 'drizzle-orm'
    import { db } from '@/db'
    import { users, verificationTokens, passwordResetTokens } from '@/db/schema'
    import { sendEmail } from '@/lib/email'
    import { verificationEmail, passwordResetEmail } from '@/lib/email-templates'

    const hash = (raw: string) => createHash('sha256').update(raw).digest('hex')
    const TTL_MS = 1000 * 60 * 60 // 1h

    export async function sendVerificationEmail(userId: string, email: string) {
      const raw = randomBytes(32).toString('hex')
      await db.insert(verificationTokens).values({
        userId, tokenHash: hash(raw), expiresAt: new Date(Date.now() + TTL_MS),
      })
      const url = `${process.env.APP_URL}/verify-email?token=${raw}`
      const { subject, html } = verificationEmail(url)
      return sendEmail({ to: email, subject, html })
    }

    export async function consumeVerificationToken(rawToken: string) {
      const row = await db.query.verificationTokens.findFirst({ where: eq(verificationTokens.tokenHash, hash(rawToken)) })
      if (!row || row.usedAt || row.expiresAt < new Date()) return null
      await db.update(verificationTokens).set({ usedAt: new Date() }).where(eq(verificationTokens.id, row.id))
      await db.update(users).set({ emailVerified: new Date() }).where(eq(users.id, row.userId))
      return row.userId
    }

    export async function requestPasswordReset(email: string) {
      const user = await db.query.users.findFirst({ where: eq(users.email, email.toLowerCase().trim()), columns: { id: true } })
      if (!user) return { data: null, error: null } // do NOT reveal whether the email exists
      const raw = randomBytes(32).toString('hex')
      await db.insert(passwordResetTokens).values({
        userId: user.id, tokenHash: hash(raw), expiresAt: new Date(Date.now() + TTL_MS),
      })
      const url = `${process.env.APP_URL}/reset-password?token=${raw}`
      const { subject, html } = passwordResetEmail(url)
      return sendEmail({ to: email, subject, html })
    }

    export async function consumeResetToken(rawToken: string) {
      const row = await db.query.passwordResetTokens.findFirst({ where: eq(passwordResetTokens.tokenHash, hash(rawToken)) })
      if (!row || row.usedAt || row.expiresAt < new Date()) return null
      await db.update(passwordResetTokens).set({ usedAt: new Date() }).where(eq(passwordResetTokens.id, row.id))
      return row.userId // caller updates the password; token already marked single-use
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser -> reset-request action | attacker may probe arbitrary emails (account enumeration) |
| email link -> token consumption | reset/verify token presented; may be forged, replayed, or expired |
| browser -> new-password form | new password crosses here; must be hashed server-side before storage |
| app -> Resend | outbound send carries the reset/verify URL |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-15 | Information Disclosure | password-reset request reveals which emails exist | mitigate | requestPasswordResetAction always returns the same generic message regardless of whether the user exists (and on parse failure) |
| T-01-16 | Spoofing | forged / guessed reset or verification token | mitigate | tokens are 32 random bytes (256-bit), stored as sha256 hash; lookup is by hash; raw token never persisted |
| T-01-17 | Tampering | reset/verify token replay after use or expiry | mitigate | consume* rejects used (usedAt set) or expired (expiresAt < now) tokens and marks usedAt on consumption (single-use) |
| T-01-18 | Information Disclosure | reset link / raw token logged or client-exposed | mitigate | email-flows.ts is server-only; the raw token lives only in the emailed URL; not logged |
| T-01-19 | Tampering | password updated without a valid token | mitigate | resetPasswordAction updates users.hashedPassword ONLY after consumeResetToken returns a userId; bad token -> no DB write (test-asserted) |
| T-01-20 | Information Disclosure | new password stored in plaintext | mitigate | bcrypt.hash(password, 10) before db.update; only the hash is persisted |
| T-01-21 | Denial of Service | reset-email flooding | accept | v1 has no per-email rate limit on reset requests; acceptable for a small internal user base; revisit in Phase 10 hardening |
</threat_model>
